	<div
		v-if="isExternal"
		:style="styleExternalIcon"
		class="svg-external-icon svg-icon"
		v-on="$listeners"
	/>
	<svg v-else :class="svgClass" aria-hidden="true" v-on="$listeners">
		<use :href="iconName" />
	</svg>

<script>
	export default {
		name: "SvgIcon",
		props: {
			iconClass: {
				type: String,
				required: true,
			},
			className: {
				type: String,
				default: "",
			},
		},
		computed: {
			iconName() {
				return `#icon-${this.iconClass}`;
			},
			svgClass() {
				if (this.className) {
					return "svg-icon " + this.className;
				} else {
					return "svg-icon";
				}
			},
		},
	};
</script>
